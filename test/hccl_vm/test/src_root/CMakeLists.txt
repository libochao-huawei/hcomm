cmake_minimum_required(VERSION 3.14)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -pthread -D_GLIBCXX_USE_CXX11_ABI=0")

if(DEFINED ENV{HCOMM_CODE_HOME})
    set(HCOMM_CODE_ROOT $ENV{HCOMM_CODE_HOME})
else()
    message(WARNING "HCOMM_CODE_HOME environment variable is not set")
endif()

if(DEFINED ENV{ASCEND_TOOLKIT_HOME})
    set(ASCEND_TOOLKIT_ROOT $ENV{ASCEND_TOOLKIT_HOME})
else()
    message(WARNING "ASCEND_TOOLKIT_HOME environment variable is not set")
endif()

include_directories(
    ${CMAKE_SOURCE_DIR}/src
    ${CMAKE_SOURCE_DIR}/src/common
    ${CMAKE_SOURCE_DIR}/src/plugin
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_SOURCE_DIR}/include/store
    ${CMAKE_SOURCE_DIR}/include/runnerdb
    ${CMAKE_BINARY_DIR}/_deps/yaml-cpp-src/include
    ${CMAKE_SOURCE_DIR}/third_party/nlohmann_json
    ${ASCEND_CANN_PACKAGE_PATH}/x86_64-linux/include
    ${ASCEND_CANN_PACKAGE_PATH}/x86_64-linux/include/hccl
    ${ASCEND_CANN_PACKAGE_PATH}/pkg_inc
    ${ASCEND_CANN_PACKAGE_PATH}/pkg_inc/hcomm/ccu
)

link_libraries(
    gtest_main
    gtest
)

# test_sim_plugin
add_executable(test_sim_plugin sim_plugin_test.cc
    ${CMAKE_SOURCE_DIR}/src/plugin/hccl_plugin.cc
)
target_link_libraries(test_sim_plugin gtest_main gtest log)
set_target_properties(test_sim_plugin PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/output/bin"
)

# test_sim_ip_address
add_executable(test_sim_ip_address sim_ip_address_test.cc)
target_include_directories(test_sim_ip_address PRIVATE
    ${ASCEND_CANN_PACKAGE_PATH}/x86_64-linux/pkg_inc
    ${ASCEND_CANN_PACKAGE_PATH}/x86_64-linux/pkg_inc/hccl
    ${ASCEND_CANN_PACKAGE_PATH}/x86_64-linux/include
)
target_link_libraries(test_sim_ip_address PRIVATE
    gtest_main
    gtest
)
set_target_properties(test_sim_ip_address PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/output/bin"
)
