function(ccu_executor_register_test TEST_NAME)
    add_test(NAME ${TEST_NAME} COMMAND ${TEST_NAME})
    set_property(TEST ${TEST_NAME} PROPERTY ENVIRONMENT
        "GCOV_PREFIX=${CMAKE_BINARY_DIR}/coverage/${TEST_NAME}"
    )
endfunction()
