add_executable(legacy_alg_ccu_1d_testcase_part2 ${test_case_list})

#头文件搜索路径
target_include_directories(legacy_alg_ccu_1d_testcase_part2 PRIVATE
    ${include_list}
)

target_compile_definitions(legacy_alg_ccu_1d_testcase_part2 PRIVATE
    -DOPEN_HCCL_TEST
    _GLIBCXX_USE_CXX11_ABI=0
    -DCCL_KERNEL_AICPU
    -DHCCL_ALG_ANALYZER_DAVID
)

target_compile_options(legacy_alg_ccu_1d_testcase_part2 PRIVATE
    -U_FORTIFY_SOURCE
    -Os
    -O0 -g
    $<$<STREQUAL:${ENABLE_ASAN},true>:-fsanitize=address -fsanitize-recover=address,all -fno-omit-frame-pointer>
)

add_dependencies(legacy_alg_ccu_1d_testcase_part2 hccl_alg_analyzer_legacy hccl_orion_alg_test hccl_alg_adaption_v2)

target_link_libraries(legacy_alg_ccu_1d_testcase_part2 PRIVATE
    $<BUILD_INTERFACE:intf_pub>
    hccl_alg_analyzer_legacy
    hccl_orion_alg_test
    hccl_alg_adaption_v2
    ${ASCEND_CANN_PACKAGE_PATH}/lib64/libc_sec.so
    $<$<STREQUAL:${ENABLE_ASAN},true>:-fsanitize=address>
    -pthread
    ${THIRD_PARTY_PATH}/protobuf/lib64/libprotobuf.so
    -lrt
    -ldl
)

target_sources(legacy_alg_ccu_1d_testcase_part2
        PUBLIC
        main.cc
        testcase_utils.cc
        testcase_all2allv_ccu.cc
        #testcase_broadcast_ccu.cc
        ${UTILS_DIR}/common/virtual_topo_stub/virtual_topo_stub.cc
