add_executable(legacy_alg_aicpu_2d_testcase ${test_case_list})
 
#头文件搜索路径
target_include_directories(legacy_alg_aicpu_2d_testcase PRIVATE
    ${include_list}
)
 
target_compile_definitions(legacy_alg_aicpu_2d_testcase PRIVATE
    -DOPEN_HCCL_TEST
    _GLIBCXX_USE_CXX11_ABI=0
    -DCCL_KERNEL_AICPU
    -DHCCL_ALG_ANALYZER_DAVID
)
 
target_compile_options(legacy_alg_aicpu_2d_testcase PRIVATE
    -U_FORTIFY_SOURCE
    -Os
    -O0 -g
    $<$<STREQUAL:${ENABLE_ASAN},true>:-fsanitize=address -fsanitize-recover=address,all -fno-omit-frame-pointer>
)

add_dependencies(legacy_alg_aicpu_2d_testcase
    hccl_alg_adaption_v2
)

target_link_libraries(legacy_alg_aicpu_2d_testcase PRIVATE
    $<BUILD_INTERFACE:intf_pub>
    hccl_alg_analyzer_legacy
    hccl_orion_alg_test
    hccl_alg_adaption_v2
    ${ASCEND_CANN_PACKAGE_PATH}/lib64/libc_sec.so
    $<$<STREQUAL:${ENABLE_ASAN},true>:-fsanitize=address>
    -pthread
    ${THIRD_PARTY_PATH}/protobuf/lib64/libprotobuf.so
    -lrt
    -ldl
)

target_sources(legacy_alg_aicpu_2d_testcase
    PUBLIC
    main.cc
    testcase_all_reduce_aicpu_2d.cc
    testcase_utils.cc
    #testcase_scatter_aicpu_2d.cc
    testcase_all_gather_aicpu_2d.cc
    testcast_alltoall_aicpu_2d.cc
    testcase_broadcast.cc
    testcase_reduce_aicpu_2d.cc
    testcase_reduce_scatter_aicpu_2d.cc
)

enable_testing()
add_test(NAME legacy_alg_aicpu_2d_testcase COMMAND legacy_alg_aicpu_2d_testcase)
