# 定义 hccd 动态链接库，在 device 侧使用
add_library(hccd SHARED)

# 宏定义
target_compile_definitions(hccd PRIVATE
    HCCD
)

# 编译选项
target_compile_options(hccd PRIVATE
    -Werror
    -Wall
    -fno-common
    -fstack-protector-strong
    -fno-strict-aliasing
    -pipe
    -O3
    -std=c++17
    $<$<CONFIG:Debug>:-g>
)

# 链接选项
target_link_options(hccd PRIVATE
    -Wl,-z,relro
    -Wl,-z,now
    -Wl,-z,noexecstack
    $<$<CONFIG:Release>:-s>
)
